package com.adobe.xml.syndication.kml
{
    import com.adobe.xml.syndication.Namespaces;
    import com.adobe.xml.syndication.ParsingTools;


    /**
    *    Class that represents an Entry element within an Atom feed
    * 
    *     @langversion ActionScript 3.0
    *    @playerversion Flash 8.5
    *    @tiptext
    * 
    *     @see http://www.atomenabled.org/developers/syndication/atom-format-spec.php#rfc.section.4.1.2
    */
    public class Overlay extends Feature
    {
        private var atom:Namespace = Namespaces.ATOM_NS;
        private var kml:Namespace = Namespaces.KML_NS;
        private var georss:Namespace = Namespaces.GEORSS_NS;

        private var _color:String;
        private var _drawOrder:Number;
        private var _icon:Icon;
        
        /**
        *    Constructor for class.
        * 
        *    @param x An XML document that contains an individual Entry element from 
        *    an Aton XML feed.
        *
        *     @langversion ActionScript 3.0
        *    @playerversion Flash 8.5
        *    @tiptext
        */    
        public function Overlay(x:XMLList)
        {
            super(x);
            this._color = ParsingTools.nullCheck(this.x.kml::color);
            this._drawOrder = ParsingTools.nanCheck(this.x.kml::drawOrder);
            if (ParsingTools.nullCheck(this.x.kml::Icon)) {
                this._icon = new Icon(this.x.kml::Icon);
            }
        }

        /**
        *    A String that contains the title for the entry.
        *
        *     @langversion ActionScript 3.0
        *    @playerversion Flash 8.5
        *    @tiptext
        */    
        public function get color():String
        {
            return this._color;
         }
        /**
        *    A String that contains the title for the entry.
        *
        *     @langversion ActionScript 3.0
        *    @playerversion Flash 8.5
        *    @tiptext
        */    
        public function get drawOrder():Number
        {
            return this._drawOrder;
         }     
         
         /**
        *    A String that contains the title for the entry.
        *
        *     @langversion ActionScript 3.0
        *    @playerversion Flash 8.5
        *    @tiptext
        */    
        public function get icon():Icon
        {
            return this._icon;
         }     
         
         public override function toString():String {
             return "Overlay: " + "color: " + this._color + "drawOrder: " + this._drawOrder    + "icon: " + this._icon;
         }
    }
}
